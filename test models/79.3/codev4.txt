import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers, regularizers
from tensorflow.keras.applications import MobileNetV2
import matplotlib.pyplot as plt
import numpy as np

# ==========================================
# 1. MOUNT GOOGLE DRIVE
# ==========================================
from google.colab import drive
drive.mount('/content/drive')

BASE_DIR  = '/content/drive/MyDrive/Chicken_Dataset'
TRAIN_DIR = f'{BASE_DIR}/Train'
VALID_DIR = f'{BASE_DIR}/Valid'
TEST_DIR  = f'{BASE_DIR}/Test'

# ==========================================
# 2. CONFIGURATION
# ==========================================
IMG_SIZE         = (320, 320)
BATCH_SIZE       = 32
NUM_CLASSES      = 3    # adulterated, edible, spoiled
EPOCHS           = 40
FINE_TUNE_EPOCHS = 20

# ==========================================
# 3. LOAD DATASETS
# ==========================================
print("Loading datasets...")

train_dataset = tf.keras.utils.image_dataset_from_directory(
    TRAIN_DIR, shuffle=True, batch_size=BATCH_SIZE,
    image_size=IMG_SIZE, label_mode='categorical'
)
val_dataset = tf.keras.utils.image_dataset_from_directory(
    VALID_DIR, shuffle=False, batch_size=BATCH_SIZE,
    image_size=IMG_SIZE, label_mode='categorical'
)
test_dataset = tf.keras.utils.image_dataset_from_directory(
    TEST_DIR, shuffle=False, batch_size=BATCH_SIZE,
    image_size=IMG_SIZE, label_mode='categorical'
)

CLASS_NAMES = train_dataset.class_names
print(f"Classes found: {CLASS_NAMES}")

AUTOTUNE      = tf.data.AUTOTUNE
train_dataset = train_dataset.cache().prefetch(AUTOTUNE)
val_dataset   = val_dataset.cache().prefetch(AUTOTUNE)
test_dataset  = test_dataset.cache().prefetch(AUTOTUNE)

# ==========================================
# 4. CLASS BALANCE + WEIGHTS
# ==========================================
print("\nChecking class distribution...")

label_counts = np.zeros(NUM_CLASSES, dtype=int)
for _, labels in train_dataset.unbatch():
    label_counts[np.argmax(labels.numpy())] += 1

print(f"Training label counts: { {CLASS_NAMES[i]: label_counts[i] for i in range(NUM_CLASSES)} }")

total = label_counts.sum()
base_weight = {
    i: total / (NUM_CLASSES * count) if count > 0 else 1.0
    for i, count in enumerate(label_counts)
}

# FIX v4 (surgical): v2 and v3 overcorrected with extreme multipliers.
# The approach here is minimal and deliberate:
#   - adulterated ×1.8: meaningful boost without destabilizing the gradient
#   - edible ×1.0:      neutral — its recall was 92%+ in v1, leave it alone
#   - spoiled ×1.3:     slight boost, it was already decent at 76.6% in v1
#
# Extreme weights (×2.2 in v3) caused the optimizer to take huge steps on
# adulterated samples, overshooting the loss surface and pulling the model
# away from the stable solution found in Stage 1.
class_weight = {
    0: base_weight[0] * 1.8,   # adulterated
    1: base_weight[1] * 1.0,   # edible
    2: base_weight[2] * 1.3    # spoiled
}
print(f"Class weights: {class_weight}")

# ==========================================
# 5. DATA AUGMENTATION
# ==========================================
# Back to a moderate set. v2's augmentation was fine; v3 didn't change it much.
# The goal is regularization, not feature destruction.
data_augmentation = keras.Sequential([
    layers.RandomFlip("horizontal_and_vertical"),
    layers.RandomRotation(0.07),
    layers.RandomZoom(0.10),
    layers.RandomTranslation(0.08, 0.08),
    layers.RandomContrast(0.15),
    layers.RandomBrightness(0.2),
], name="augmentation")

# ==========================================
# 6. BUILD MODEL
# ==========================================
base_model = MobileNetV2(
    input_shape=(320, 320, 3),
    include_top=False,
    weights='imagenet'
)
base_model.trainable = False

inputs = keras.Input(shape=(320, 320, 3))
x      = data_augmentation(inputs)
x      = layers.Rescaling(scale=1./127.5, offset=-1)(x)
x      = base_model(x, training=False)
x      = layers.GlobalAveragePooling2D()(x)
x      = layers.Dense(256, activation='relu',
                      kernel_regularizer=regularizers.l2(1e-4))(x)
x      = layers.Dropout(0.5)(x)
outputs = layers.Dense(NUM_CLASSES, activation='softmax',
                       kernel_regularizer=regularizers.l2(1e-4))(x)

model = keras.Model(inputs, outputs)
model.summary()

# ==========================================
# 7. STAGE 1 — COMPILE + TRAIN (FROZEN BASE)
# ==========================================
# Back to CategoricalCrossentropy with label_smoothing=0.1.
# Focal loss (v3) introduced training instability when combined with high
# class weights. Standard CE + label smoothing was more stable in v1.
model.compile(
    optimizer=keras.optimizers.Adam(learning_rate=1e-4),
    loss=keras.losses.CategoricalCrossentropy(label_smoothing=0.1),
    metrics=[
        'accuracy',
        keras.metrics.TopKCategoricalAccuracy(k=2),
        keras.metrics.Precision(name='precision'),
        keras.metrics.Recall(name='recall'),
    ]
)

callbacks_stage1 = [
    keras.callbacks.EarlyStopping(
        monitor='val_loss', patience=5, restore_best_weights=True
    ),
    keras.callbacks.ReduceLROnPlateau(
        monitor='val_loss', factor=0.3, patience=3, min_lr=1e-7
    ),
]

print("\nStarting Stage 1 training (frozen base)...")
history = model.fit(
    train_dataset,
    validation_data=val_dataset,
    epochs=EPOCHS,
    callbacks=callbacks_stage1,
    class_weight=class_weight
)

# ==========================================
# 8. FINE-TUNING (STAGE 2) — STABILIZED
# ==========================================
# This is where v2 and v3 both failed. Three targeted fixes:
#
# FIX A — Unfreeze only 30 layers (not 50 or 80).
#   The loss spikes and val accuracy collapses in v2/v3 show that unfreezing
#   too many layers at once disrupts well-learned features. 30 layers exposes
#   just the high-level feature detectors near the top of the backbone —
#   enough to adapt to chicken-specific patterns without catastrophic forgetting.
#
# FIX B — Explicitly freeze ALL BatchNorm layers backbone-wide.
#   Even with base_model(x, training=False), when trainable=True the BN layers
#   can still receive gradient updates that shift their running stats. Locking
#   them down prevents the feature distribution from shifting mid-fine-tune.
#
# FIX C — gradient clipping (clipnorm=1.0) + very low LR (1e-6).
#   The v3 loss spike at fine-tune start reached 0.75 — a sign of exploding
#   gradients on newly unfrozen weights. Clipping caps the gradient norm so
#   the first few fine-tuning epochs don't undo Stage 1's work.

print("\nStarting Stage 2 fine-tuning (unfreezing last 30 layers, frozen BN)...")

base_model.trainable = True

# Unfreeze only the last 30 layers
for layer in base_model.layers[:-30]:
    layer.trainable = False

# Explicitly freeze ALL BatchNorm layers regardless of position
for layer in base_model.layers:
    if isinstance(layer, layers.BatchNormalization):
        layer.trainable = False

n_trainable = sum(l.trainable for l in base_model.layers)
print(f"Trainable layers in backbone: {n_trainable} / {len(base_model.layers)}")

callbacks_stage2 = [
    keras.callbacks.EarlyStopping(
        monitor='val_loss', patience=6, restore_best_weights=True
    ),
    keras.callbacks.ReduceLROnPlateau(
        monitor='val_loss', factor=0.3, patience=3, min_lr=1e-8
    ),
]

model.compile(
    optimizer=keras.optimizers.Adam(
        learning_rate=1e-6,   # much lower than v3's 5e-6
        clipnorm=1.0          # gradient clipping — prevents loss spike
    ),
    loss=keras.losses.CategoricalCrossentropy(label_smoothing=0.1),
    metrics=[
        'accuracy',
        keras.metrics.TopKCategoricalAccuracy(k=2),
        keras.metrics.Precision(name='precision'),
        keras.metrics.Recall(name='recall'),
    ]
)

history_fine = model.fit(
    train_dataset,
    validation_data=val_dataset,
    epochs=FINE_TUNE_EPOCHS,
    callbacks=callbacks_stage2,
    class_weight=class_weight
)

# ==========================================
# 9. PLOT COMBINED TRAINING HISTORY
# ==========================================
acc      = history.history['accuracy']      + history_fine.history['accuracy']
val_acc  = history.history['val_accuracy']  + history_fine.history['val_accuracy']
loss     = history.history['loss']          + history_fine.history['loss']
val_loss = history.history['val_loss']      + history_fine.history['val_loss']
prec     = history.history['precision']     + history_fine.history['precision']
rec      = history.history['recall']        + history_fine.history['recall']

stage1_len = len(history.history['accuracy'])

fig, axes = plt.subplots(1, 3, figsize=(20, 5))
plot_cfg = [
    (acc,  val_acc,  'Accuracy',               'Train Acc',  'Val Acc'),
    (loss, val_loss, 'Loss',                   'Train Loss', 'Val Loss'),
    (prec, rec,      'Train Precision/Recall', 'Precision',  'Recall'),
]
for ax, (a, b, title, la, lb) in zip(axes, plot_cfg):
    ax.plot(a, label=la)
    ax.plot(b, label=lb)
    ax.axvline(x=stage1_len, color='gray', linestyle='--', label='Fine-tune start')
    ax.set_title(title)
    ax.set_xlabel('Epoch')
    ax.legend()

plt.tight_layout()
plt.savefig(f'{BASE_DIR}/training_history_v4.png', dpi=150)
plt.show()
print("Training plot saved.")

# ==========================================
# 10. FINAL EVALUATION ON TEST SET
# ==========================================
print("\nFinal evaluation on held-out test set:")
results = model.evaluate(test_dataset)
print(f"Test Accuracy  : {results[1]:.4f}")
print(f"Top-2 Accuracy : {results[2]:.4f}")
print(f"Precision      : {results[3]:.4f}")
print(f"Recall         : {results[4]:.4f}")
print(f"Test Loss      : {results[0]:.4f}")

# ==========================================
# 11. CONFUSION MATRIX & CLASSIFICATION REPORT
# ==========================================
from sklearn.metrics import confusion_matrix, classification_report, precision_recall_fscore_support
import seaborn as sns

print("\nGenerating predictions for the confusion matrix...")

y_pred_probs = model.predict(test_dataset)
y_pred       = np.argmax(y_pred_probs, axis=1)
y_true       = np.concatenate([
    np.argmax(labels.numpy(), axis=1) for _, labels in test_dataset
])

cm = confusion_matrix(y_true, y_pred)

plt.figure(figsize=(8, 6))
sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
            xticklabels=CLASS_NAMES, yticklabels=CLASS_NAMES, cbar=False)
plt.title('Confusion Matrix (Test Set)', fontsize=14, pad=15)
plt.ylabel('True Label',      fontsize=12)
plt.xlabel('Predicted Label', fontsize=12)
plt.xticks(rotation=45)
plt.yticks(rotation=0)
plt.tight_layout()
plt.savefig(f'{BASE_DIR}/confusion_matrix_v4.png', dpi=150)
plt.show()
print("Confusion matrix saved.")

print("\n" + "="*50)
print(" CLASSIFICATION REPORT")
print("="*50)
print(classification_report(y_true, y_pred, target_names=CLASS_NAMES))

# ==========================================
# 12. PER-CLASS CONFIDENCE DISTRIBUTIONS
# ==========================================
# Health check: "True: adulterated" P(adulterated) should now be
# right-skewed. If it's still flat, the adulterated class needs
# more training data — no further architecture changes will fix it.
fig, axes = plt.subplots(1, NUM_CLASSES, figsize=(15, 4), sharey=True)
for true_class_idx, ax in enumerate(axes):
    mask        = y_true == true_class_idx
    class_probs = y_pred_probs[mask]
    for pred_class_idx, class_name in enumerate(CLASS_NAMES):
        ax.hist(class_probs[:, pred_class_idx], bins=20, alpha=0.6,
                label=f'P({class_name})')
    ax.set_title(f'True: {CLASS_NAMES[true_class_idx]}')
    ax.set_xlabel('Predicted Probability')
    ax.legend(fontsize=7)
axes[0].set_ylabel('Count')
plt.suptitle('Per-Class Confidence Distributions (Test Set)', fontsize=13)
plt.tight_layout()
plt.savefig(f'{BASE_DIR}/confidence_distributions_v4.png', dpi=150)
plt.show()

# ==========================================
# 13. PER-CLASS METRICS SUMMARY TABLE
# ==========================================
prec_arr, rec_arr, f1_arr, sup_arr = precision_recall_fscore_support(
    y_true, y_pred, labels=[0, 1, 2]
)
overall_acc = np.sum(y_pred == y_true) / len(y_true)

print("\n" + "="*60)
print(f" {'Class':<15} {'Precision':>10} {'Recall':>10} {'F1':>8} {'Support':>9}")
print("="*60)
for i, name in enumerate(CLASS_NAMES):
    print(f" {name:<15} {prec_arr[i]:>9.1%} {rec_arr[i]:>9.1%} "
          f"{f1_arr[i]:>7.1%} {int(sup_arr[i]):>9}")
print("="*60)
print(f" {'Overall Accuracy':<15} {overall_acc:>9.1%}")
print("="*60)
